const queryArray = (array, key, keyAlias = 'key') => {
  if (!(array instanceof Array)) {
    return null
  }
  let data
 
  for (const item of array) {
    if (item[keyAlias] === key) {
      data = item
      break
    }
  }
 
  if (data) {
    return data
  }
  return null
}
